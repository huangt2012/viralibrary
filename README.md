# veralibrary
# 安装最后已经发布的组件库来试试
npm install viralibrary --save

# 使用

// 加载样式
import 'viralibrary/dist/index.css'
// 引入组件
import { Button } from 'viralibrary'


# 一些本地开发命令
//启动本地环境
npm run stroybook

//跑单元测试
npm test

//build可发布静态文件
npm run build

//发布到 npm
npm run publish
